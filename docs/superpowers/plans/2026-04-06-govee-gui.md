# Govee GUI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone Iced GUI app for controlling Govee LED strips, sharing the existing `govee-lan` library via a Cargo workspace.

**Architecture:** Cargo workspace with three crates (`govee-lan`, `govee-cli`, `govee-gui`). The GUI uses `govee-lan` directly for simple commands (on/off/color/brightness) and shells out to the `govee` CLI binary for continuous modes (screen/audio/ambient/themes). Theme data types and loaders move into `govee-lan` so the GUI can render the theme card grid.

**Tech Stack:** Rust, Iced 0.13 (wgpu backend), tokio, serde, toml, dirs

**Spec:** `docs/superpowers/specs/2026-04-06-govee-gui-design.md`

/// Sample a color from a palette at position t (0.0-1.0).
pub fn palette_sample(anchors: &[PA], t: f64) -> Rgb {
    if anchors.is_empty() {
        return (0, 0, 0);
    }
    if anchors.len() == 1 {
        return (anchors[0].r, anchors[0].g, anchors[0].b);
    }
    let t = t.clamp(0.0, 1.0);
    if t <= anchors[0].pos {
        return (anchors[0].r, anchors[0].g, anchors[0].b);
    }
    for i in 0..anchors.len() - 1 {
        let a = &anchors[i];
        let b = &anchors[i + 1];
        if t <= b.pos {
            let f = if (b.pos - a.pos).abs() < 1e-9 {
                0.0
            } else {
                ((t - a.pos) / (b.pos - a.pos)).clamp(0.0, 1.0)
            };
            return (
                (a.r as f64 + (b.r as f64 - a.r as f64) * f) as u8,
                (a.g as f64 + (b.g as f64 - a.g as f64) * f) as u8,
                (a.b as f64 + (b.b as f64 - a.b as f64) * f) as u8,
            );
        }
    }
    let l = anchors.last().unwrap();
    (l.r, l.g, l.b)
}

pub fn lerp_rgb(a: Rgb, b: Rgb, t: f64) -> Rgb {
    let t = t.clamp(0.0, 1.0);
    (
        (a.0 as f64 + (b.0 as f64 - a.0 as f64) * t) as u8,
        (a.1 as f64 + (b.1 as f64 - a.1 as f64) * t) as u8,
        (a.2 as f64 + (b.2 as f64 - a.2 as f64) * t) as u8,
    )
}

pub fn hsv_to_rgb(h: f64, s: f64, v: f64) -> Rgb {
    let h = (h.fract() + 1.0).fract() * 6.0;
    let f = h.fract();
    let p = v * (1.0 - s);
    let q = v * (1.0 - s * f);
    let tv = v * (1.0 - s * (1.0 - f));
    let (r, g, b) = match h as u32 {
        0 => (v, tv, p),
        1 => (q, v, p),
        2 => (p, v, tv),
        3 => (p, q, v),
        4 => (tv, p, v),
        _ => (v, p, q),
    };
    ((r * 255.0) as u8, (g * 255.0) as u8, (b * 255.0) as u8)
}

```rust
use std::fs;
use std::path::PathBuf;

use crate::theme_defs::builtin_themes;
use crate::themes::ThemeDef;

fn user_themes_dir() -> Option<PathBuf> {
    dirs::config_dir().map(|d| d.join("govee").join("themes"))
}

fn load_user_themes() -> Vec<ThemeDef> {
    let dir = match user_themes_dir() {
        Some(d) if d.is_dir() => d,
        _ => return vec![],
    };

    let mut themes = Vec::new();
    let entries = match fs::read_dir(&dir) {
        Ok(e) => e,
        Err(_) => return vec![],
    };

    for entry in entries.flatten() {
        let path = entry.path();
        if path.extension().is_some_and(|e| e == "toml") {
            if let Ok(content) = fs::read_to_string(&path) {
                if let Ok(theme) = toml::from_str::<ThemeDef>(&content) {
                    themes.push(theme);
                }
            }
        }
    }

    themes
}

pub fn load_all_themes() -> Vec<ThemeDef> {
    let mut themes = builtin_themes();
    for ut in load_user_themes() {
        if let Some(pos) = themes.iter().position(|t| t.name == ut.name) {
            themes[pos] = ut;
        } else {
            themes.push(ut);
        }
    }
    themes
}

use iced::widget::{column, container, text};
use iced::{Element, Length, Size, Task, Theme};

fn main() -> iced::Result {
    iced::application("Govee", App::update, App::view)
        .theme(|_| {
            Theme::custom(
                "Govee Dark".into(),
                iced::theme::Palette {
                    background: style::BG,

        .run_with(App::new)
}

struct App;

#[derive(Debug, Clone)]
enum Message {}

impl App {
    fn new() -> (Self, Task<Message>) {
        (App, Task::none())
    }

    fn update(&mut self, _message: Message) -> Task<Message> {
        Task::none()
    }

    fn view(&self) -> Element<Message> {
        let content = column![text("Govee GUI").size(24),]
            .spacing(style::SPACING);

    if let Some(ip) = device_ip {
        cmd.arg("--ip").arg(ip);
    }
    cmd.args(args);
    cmd.spawn()
}

/// Kill a running subprocess gracefully.
pub fn kill(child: &mut Child) {
    // Send SIGTERM on Unix for graceful DreamView deactivation
    #[cfg(unix)]
    {
        unsafe {
            libc::kill(child.id() as i32, libc::SIGTERM);
        }
    }
    #[cfg(not(unix))]
    {
        let _ = child.kill();
    }
}

    // Kill any running subprocess first
    if let Some(ref mut child) = self.subprocess {
        crate::subprocess::kill(child);
        self.subprocess = None;
    }

fn theme_card(theme: &govee_lan::ThemeDef, is_active: bool) -> Element<'static, Message> {
    let name = theme.name.clone();
    let is_animated = matches!(theme.kind, govee_lan::ThemeKind::Animated { .. });

    // Build color band based on theme kind
    let color_band = match &theme.kind {
        govee_lan::ThemeKind::Solid { color } => {
            container(text(""))
                .width(Length::Fill)
                .height(32)
                .style(move |_theme| container::Style {
                    background: Some(iced::Background::Color(iced::Color::from_rgb8(
                        color.0, color.1, color.2,
                    ))),
                    ..Default::default()
                })
        }
        govee_lan::ThemeKind::Animated { .. } => {
            // Use accent color as placeholder for animated themes
            container(text(""))
                .width(Length::Fill)
                .height(32)
                .style(|_theme| container::Style {
                    background: Some(iced::Background::Color(style::SURFACE)),
                    ..Default::default()
                })
        }
    };

    let mut info = column![text(theme.name.clone()).size(13)].spacing(2);
    if is_animated {
        info = info.push(text("Animated").size(11).color(style::TEXT_MUTED));
    }
    if is_active {
        info = info.push(text("● Active").size(11).color(style::SUCCESS));
    }

    let card = button(
        column![color_band, container(info).padding([6, 8])]
    )
    .on_press(Message::ApplyTheme(name))
    .padding(0)
    .width(Length::Fixed(140.0));

    if is_active {
        container(card)
            .style(|_theme| container::Style {
                border: iced::Border {
                    color: style::ACCENT,
                    width: 2.0,
                    radius: style::RADIUS.into(),
                },
                ..Default::default()
            })

pub fn view(app: &App) -> Element<Message> {
    // Filter tabs
    let tabs: Vec<Element<Message>> = CATEGORIES
        .iter()
        .map(|&cat| category_tab(cat, app.theme_filter == cat))
        .collect();

    // Check for user themes
    let has_user = app.themes.iter().any(|t| {
        !CATEGORIES[1..].contains(&t.category.as_str())
    });
    let mut tab_row = Row::with_children(tabs).spacing(4);
    if has_user {
        tab_row = tab_row.push(category_tab("user", app.theme_filter == "user"));
    }

    let header = row![
        text("Themes").size(20),
        iced::widget::horizontal_space(),
        tab_row,
    ]
    .align_y(iced::Alignment::Center);

    // Group themes by category
    let filtered: Vec<&govee_lan::ThemeDef> = if app.theme_filter == "all" {
        app.themes.iter().collect()
    } else {
        app.themes.iter().filter(|t| t.category == app.theme_filter).collect()
    };

    // Build category sections
    let mut content = column![header].spacing(16);

    let categories_to_show: Vec<&str> = if app.theme_filter == "all" {
        let mut cats: Vec<&str> = vec![];
        for t in &filtered {
            if !cats.contains(&t.category.as_str()) {
                cats.push(&t.category);
            }
        }
        cats
    } else {
        vec![&app.theme_filter]
    };

    for cat in categories_to_show {
        let cat_themes: Vec<&govee_lan::ThemeDef> = filtered
            .iter()
            .filter(|t| t.category == cat)
            .copied()
            .collect();

        if cat_themes.is_empty() {
            continue;
        }

        let label = text(cat.to_uppercase()).size(11).color(style::TEXT_MUTED);

        let cards: Vec<Element<Message>> = cat_themes
            .iter()
            .map(|t| {
                let active = app.active_theme.as_deref() == Some(&t.name);
                theme_card(t, active)
            })
            .collect();

        // Wrap cards into rows of ~4
        let grid = iced::widget::wrap::Wrap::with_elements(cards)
            .spacing(8.0)
            .line_spacing(8.0);

        content = content.push(column![label, grid].spacing(8));
    }

    scrollable(content)
        .width(Length::Fill)
        .height(Length::Fill)
        .into()
}
```

Note: Iced 0.13 may not have `wrap` built-in. If not available, use a series of `Row` widgets or the `iced_aw` crate's wrap. Adapt as needed during implementation. A simpler fallback is just putting all cards in a `Column` of `Row`s.
